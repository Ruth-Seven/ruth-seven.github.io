快速幂算法，也称二分幂。 递归写法：



```
typedef long long ll;
ll binaryPow(ll a, ll b, ll m){
	if(b == 0) return 1;
	if(b & 1) return a * binaryPow(a, b - 1, m) % m;
	else{
		ll mul = binaryPow(a, b / 2, m);
		return mul * mul % m;
	}
}
```

迭代写法：

```
typedef long long ll;
ll binaryPow(ll a, ll b, ll m){
	ll res = 1， pow = a;
	while(b){
		if(b & 1)
			res = res * pow % m; //注意这两个地方的取余
		pow = pow * pow % m; 
		b >> 1;
	}
	return res;
}
```
