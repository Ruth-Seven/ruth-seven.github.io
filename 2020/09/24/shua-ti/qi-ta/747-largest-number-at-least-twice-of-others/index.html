747. Largest Number At Least Twice of Others

747. Largest Number At Least Twice of Others

EZ

思路：

沙雕题目。

代码：

class Solution {
public:
    int dominantIndex(vector<int>& nums) {
        int numbig = -1, numlitter= -1, idx = -1;
        if(nums.size() == 1) return 0;
        for(int i = 0; i < nums.size(); i++){
            if(nums[i] > numbig){
                idx = i;
                numlitter = numbig;
                numbig = nums[i];
            }
            else if(nums[i] > numlitter){
                numlitter = nums[i];
            }
        }
        if(numbig >= numlitter * 2 ){
            return idx;
        }
        else return -1;
    }
};
